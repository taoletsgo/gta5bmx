复制代码

用记事本新建一个json-web-search，后缀改为.py，复制以下内容保存

import requests
import json

def crawl_urls(base_url):
    for i in range(3):
        for j in range(3):
            for lang in ["en", "ja", "zh", "zh-cn", "fr", "de", "it", "ru", "pt", "pl", "pt", "kr", "es", "es-mx"]:
                old_url = f"{base_url}/2_0.jpg"
                url = f"{base_url.rsplit('/', 1)[0]}/{i}_{j}_{lang}.json"
                headers = {
                    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
                }

                try:
                    response = requests.get(url, headers=headers)

                    if response.status_code == 200:
                        try:
                            data = response.json()
                            # 检查返回的数据，根据具体需求进行处理
                            if data:
                                print(f"Valid URL: {url}")
                                input("数据已找到，请复制到Nicko's Job Editor里。按 Enter 键继续...")
                                return True  # 找到数据后返回True
                        except json.JSONDecodeError:
                            print(f"Invalid JSON response for URL: {url}")
                    elif response.status_code == 404:
                        print(f"URL not found: {url}")
                    else:
                        print(f"Error for URL {url}: {response.status_code}")

                except requests.RequestException as e:
                    print(f"Error for URL {url}: {str(e)}")

    return False  # 如果没有找到数据，返回False

if __name__ == "__main__":
    base_url = input("请输入差事图片URL（例如https://prod.cloud.rockstargames.com/ugc/gta5mission/1966/b5qPg2UtVkyK90ynmAl2-A/2_0.jpg）: ")
    
    if not crawl_urls(base_url):
        print("未找到任何数据。按 Enter 键退出。")
        input()
